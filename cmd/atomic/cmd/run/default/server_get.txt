package main

import (
	"bufio"
	"encoding/json"
	"fmt"
	"log"
	"net/http"
	"os"
)

// ===== Panic Recovery Middleware =====
func recoveryMiddleware(next http.HandlerFunc) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		defer func() {
			if rec := recover(); rec != nil {
				log.Printf("recovered from panic: %v", rec)
				http.Error(w, "internal server error", http.StatusInternalServerError)
			}
		}()
		next(w, r)
	}
}

// ===== Main =====
func main() {
	port := os.Getenv("PORT")
	if port == "" {
		port = "3000" // sensible local default
	}

	logFile, err := os.OpenFile("atomic_logs.txt", os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0644)
    if err != nil {
        fmt.Println("Failed to open log file:", err)
        return
    }
    defer logFile.Close()

    r, w, _ := os.Pipe()
    os.Stdout = w
    os.Stderr = w

    go func() {
        scanner := bufio.NewScanner(r)
        for scanner.Scan() {
            line := scanner.Text()
            // Append to file
            fmt.Fprintln(logFile, line)
        }
        if err := scanner.Err(); err != nil {
            fmt.Println("Error reading from pipe:", err)
        }
    }()

	mux := http.NewServeMux()

	mux.HandleFunc("{{ROUTE}}", recoveryMiddleware(func(w http.ResponseWriter, r *http.Request) {
		defer r.Body.Close()

		status, message, data := {{FUNC}}()

		w.Header().Set("Content-Type", "application/json")
		w.WriteHeader(status)
		_ = json.NewEncoder(w).Encode(map[string]interface{}{
			"status":  status,
			"message": message,
			"payload": data,
		})
	}))

	server := &http.Server{
		Addr:    fmt.Sprintf(":%s", port),
		Handler: mux,
	}

	log.Printf("🚀 Starting server on http://localhost:%s", port)
	log.Fatal(server.ListenAndServe())
}
